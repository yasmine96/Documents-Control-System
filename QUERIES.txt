CREATE DATABASE DOCUMENT_DB;


CREATE TABLE DOCUMENT(
ID           INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
SUBJECT      VARCHAR(30) ,
TYPE	     VARCHAR(15),
DATE         DATETIME,
TIME         TIMESTAMP,
IN_OUT       VARCHAR(3),
NO_OF_COPIES INT,
);

CREATE TABLE DRAFT(
ID           INT NOT NULL,
DRAFT_NO     INT,
SUBJECT      VARCHAR(30) ,
TYPE	     VARCHAR(15),
DATE         DATETIME,
TIME         TIMESTAMP,
IN_OUT       VARCHAR(3),
NO_OF_COPIES INT,
PRIMARY KEY(ID,DRAFT_NO.)
);


CREATE TABLE DOC_OWNER(
ID      INT AUTO_INCREMENT PRIMARY KEY,
NAME    VARCHAR(50),
PHONE   VARCHAR(15),
ADDRESS VARCHAR(30)
);

CREATE TABLE EDITOR(
ID       INT AUTO_INCREMENT PRIMARY KEY,
NAME     VARCHAR(50),
PHONE    VARCHAR(15),
ADDRESS  VARCHAR(30),
USERNAME VARCHAR(30),
PASSWORD VARCHAR(30),
ROLE 	 VARCHAR(10),
STATUS   VARCHAR(10)
);

CREATE TABLE TRACKNG_RECORD(
USER_ID      INT FOREIGN KEY REFERENCES EDITOR(ID),
TRACKING_NO  INT AUTO_INCREMENT PRIMARY KEY,
PHONE    VARCHAR(15),
ADDRESS  VARCHAR(30),
USERNAME VARCHAR(30),
PASSWORD VARCHAR(30),
ROLE 	 VARCHAR(10),
STATUS   VARCHAR(10)
);
